00042   /*
00043    * A replacement to boost::rounded_arith_opp as it has a bug under clang :
00044    *   - the to_int call made in the boost version needed to be replaced by 
00045    *     Rounding::to_int to allow the compiler know that this static method 
00046    *     is coming from the base class 
00047    */
00048   template<class T, class Rounding>
00049   struct my_rounded_arith_opp: Rounding {
00050     void init() { this->upward(); }
00051 # define BOOST_DN(EXPR) \
00052 this->downward(); \
00053 T r = this->force_rounding(EXPR); \
00054 this->upward(); \
00055 return r
00056 # define BOOST_NR(EXPR) \
00057 this->to_nearest(); \
00058 T r = this->force_rounding(EXPR); \
00059 this->upward(); \
00060 return r
00061 # define BOOST_UP(EXPR) return this->force_rounding(EXPR)
00062 # define BOOST_UP_NEG(EXPR) return -this->force_rounding(EXPR)
00063     template<class U> T conv_down(U const &v) { BOOST_UP_NEG(-v); }
00064     template<class U> T conv_up  (U const &v) { BOOST_UP(v); }
00065     T add_down(const T& x, const T& y) { BOOST_UP_NEG((-x) - y); }
00066     T sub_down(const T& x, const T& y) { BOOST_UP_NEG(y - x); }
00067     T mul_down(const T& x, const T& y) { BOOST_UP_NEG(x * (-y)); }
00068     T div_down(const T& x, const T& y) { BOOST_UP_NEG(x / (-y)); }
00069     T add_up  (const T& x, const T& y) { BOOST_UP(x + y); }
00070     T sub_up  (const T& x, const T& y) { BOOST_UP(x - y); }
00071     T mul_up  (const T& x, const T& y) { BOOST_UP(x * y); }
00072     T div_up  (const T& x, const T& y) { BOOST_UP(x / y); }
00073     T median  (const T& x, const T& y) { BOOST_NR((x + y) / 2); }
00074     T sqrt_down(const T& x)
00075     { BOOST_NUMERIC_INTERVAL_using_math(sqrt); BOOST_DN(sqrt(x)); }
00076     T sqrt_up  (const T& x)
00077     { BOOST_NUMERIC_INTERVAL_using_math(sqrt); BOOST_UP(sqrt(x)); }
00078     T int_down(const T& x) { return -Rounding::to_int(-x); }
00079     T int_up  (const T& x) { return Rounding::to_int(x); }
00080 # undef BOOST_DN
00081 # undef BOOST_NR
00082 # undef BOOST_UP
00083 # undef BOOST_UP_NEG
00084   };

00098   boost_flt convert(EUROPA::Domain const &dom) {
00099     EUROPA::edouble lo, hi;
00100     EUROPA::edouble::basis_type b_lo = boost_flt::whole().lower(), 
00101                b_hi = boost_flt::whole().upper();
00102     dom.getBounds(lo, hi);
00103     
00104     if( lo>std::numeric_limits<EUROPA::edouble>::minus_infinity() )
00105       b_lo = cast_basis(lo);
00106     if( hi<std::numeric_limits<EUROPA::edouble>::infinity() )
00107       b_hi = cast_basis(hi);
00108     return boost_flt(b_lo, b_hi);
00109   }
