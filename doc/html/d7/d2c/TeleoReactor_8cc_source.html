00299 double TeleoReactor::workRatio() {
00300   try {
00301     if( hasWork() ) {
00302       double ret = m_deadline;
00303       ret -= getCurrentTick();
00304       if( ret<=0.0 && m_nSteps>0 ) {
00305         if( !m_past_deadline ) {
00306           m_past_deadline = true;
00307           m_validSteps = m_nSteps;
00308           syslog("WARN")<<" Reactor is now exceeding its deliberation latency ("
00309           <<getLatency()<<")\n\tNumber of steps within its latency: "<<m_validSteps;
00310         }
00311         ret = m_nSteps+1;
00312       } else {
00313         ret += 1.0;
00314         ret *= m_nSteps+1;
00315       }
00316       return 1.0/ret;
00317     } else {
00318       // Dispatched goals management
00319       details::external i = ext_begin();
00320       details::goal_queue dispatched; // store the goals that got dispatched on this tick ...
00321                                       // I do nothing with it for now
00322       
00323       // Manage goal dispatching
00324       for( ; i.valid(); ++i )
00325        i.dispatch(getCurrentTick()+1, dispatched);
00326        
00327     }
00328   } catch(std::exception const &se) {
00329     syslog("WARN")<<"Exception during hasWork question: "<<se.what();
00330   } catch(...) {
00331     syslog("WARN")<<"Unknown Exception during hasWork question";
00332   }
00333   if( m_past_deadline ) {
00334     syslog("WARN")<<"Reactor needed to deliberate "<<(m_nSteps-m_validSteps)
00335                   <<" extra steps spread other "<<(getCurrentTick()-m_deadline)
00336                   <<" ticks after its latency."; 
00337   }
00338   reset_deadline();

00340 }

00634 void TeleoReactor::latency_updated(TICK old_l, TICK new_l) {
00635   TICK prev = m_maxDelay;
00636 
00637   if( new_l>m_maxDelay )
00638     m_maxDelay = new_l;
00639   else if( old_l==m_maxDelay ) {
00640     // special case : the updated value is smaller and the old value is
00641     //                equal to my former exec delay
00642     //                this means that on of the timelines that were
00643     //                constraining my maxDelay has just reduced its latency
00644     m_maxDelay = new_l;
00645     for(details::active_external i(ext_begin(), ext_end()), endi(ext_end()); endi!=i; ++i) 
00646       m_maxDelay = std::max(m_maxDelay, i->latency());
00647   }
00648   if( m_maxDelay!=prev ) {
00649     // It may be anoying on the long run but for now I will log when this
00650     // exec latency changes
00651     syslog()<<" Execution latency updated from "<<prev<<" to "<<m_maxDelay;
00652     // Notify all the reactors that depend on me
00653     for(internal_set::iterator i=m_internals.begin(); m_internals.end()!=i; ++i)
00654       (*i)->latency_update(getLatency()+prev);
00655   }
00656 }
