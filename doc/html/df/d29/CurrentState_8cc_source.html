00088   :m_assembly(assembly), m_client(assembly.plan_db()->getClient()), 
00089    m_timeline(timeline), m_id(this) {
00090   assembly.predicates(timeline, m_pred_names);
00091   // removed special values 
00092   m_pred_names.erase(Assembly::FAILED_PRED); 
00093   m_pred_names.erase(Assembly::UNDEFINED_PRED);
00094 
00095   // Now extract the default predicate (if any)
00096   EUROPA::ConstrainedVariableId default_attr = assembly.default_pred(timeline);
00097   
00098   // For now I keep this default behavior that prohibits to have no default 
00099   // It may change in the future 
00100   if( !default_attr->lastDomain().isSingleton() )
00101     throw EuropaException("Agent timeline "+timeline->getName().toString()+
00102                        " does not specify its default predicate.");
00103   m_pred_default = default_attr->lastDomain().getSingletonValue();
00104 
00105   // Now I need to check that this rpedicate is valid
00106   if( !assembly.schema()->isPredicate(predicate_name(timeline, m_pred_default)) ) 
00107     throw EuropaException("Agent timeline "+timeline->getName().toString()+
00108                        " default predicate \""+m_pred_default.toString()
00109                        +"\" does not exists.");
00110   // Finally remove the default from the alternate choices
00111   m_pred_names.erase(m_pred_default);

00236 bool CurrentState::commit() { 
00237   if( check_committed() )
00238     return true; 
00239   else {
00240     EUROPA::eint start_time;
00241     
00242     if( m_prev_obs.isId() ) {
00243       EUROPA::TokenId active = m_prev_obs;
00244       EUROPA::IntervalIntDomain i_now(now());
00245       
00246       debugMsg("trex:commit", "Terminating "<<timeline()->toString()<<'.'
00247                <<m_prev_obs->getUnqualifiedPredicateName().toString()<<'('<<m_prev_obs->getKey()<<')');
00248 
00249       if( m_prev_obs->isMerged() )
00250         active=m_prev_obs->getActiveToken();
00251       
00252       if( active->end()->lastDomain().intersects(i_now) ) {
00253         EUROPA::eint start = m_prev_obs->start()->getSpecifiedValue();
00254         
00255         restrict_base(m_prev_obs, m_prev_obs->end(), EUROPA::IntervalIntDomain(now()));
00256         restrict_base(m_prev_obs, m_prev_obs->duration(), EUROPA::IntervalIntDomain(now()-start));
00257         
00258         m_assembly.terminate(m_prev_obs);
00259         m_prev_obs = EUROPA::TokenId::noId();        
00260       } else {
00261         debugMsg("trex:always", "Unexpected termination of token "<<m_prev_obs->toString());
00262         return false;
00263       }
00264       debugMsg("trex:commit", "Starting "<<timeline()->toString()<<'.'
00265                <<m_last_obs->getUnqualifiedPredicateName().toString()<<'('<<m_last_obs->getKey()<<')');
00266       restrict_base(m_last_obs, m_last_obs->start(), EUROPA::IntervalIntDomain(now()));
00267       m_assembly.notify(*this);
00268       if( m_constraint.isId() ) {
00269         m_client->deleteConstraint(m_constraint);
00270         m_constraint = EUROPA::ConstraintId::noId();
00271       }
00272       start_time = now();
00273     } else
00274       start_time = m_last_obs->start()->getSpecifiedValue();
00275       
00276     if( now()==start_time ) {
00277       debugMsg("trex:commit", "Starting "<<timeline()->toString()<<'.'
00278                <<m_last_obs->getUnqualifiedPredicateName().toString()<<'('<<m_last_obs->getKey()<<')');
00279       restrict_base(m_last_obs, m_last_obs->start(), EUROPA::IntervalIntDomain(now()));
00280       m_assembly.notify(*this);      
00281     } else {
00282       EUROPA::IntervalIntDomain future(now()+1, std::numeric_limits<EUROPA::eint>::infinity()),
00283         dur(now()+1-start_time, std::numeric_limits<EUROPA::eint>::infinity());
00284     
00285       debugMsg("trex:commit", "Extend duration of"<<timeline()->toString()<<'.'
00286                <<m_last_obs->getUnqualifiedPredicateName().toString()<<'('<<m_last_obs->getKey()<<')');
00287       restrict_base(m_last_obs, m_last_obs->end(), future);
00288       restrict_base(m_last_obs, m_last_obs->duration(), dur);
00289     }
00290     if( m_constraint.isId() ) {
00291       m_assembly.plan_db()->getClient()->deleteConstraint(m_constraint);
00292       m_constraint = EUROPA::ConstraintId::noId();
00293     }
00294     return m_assembly.constraint_engine()->propagate();
00295   }

00381 void CurrentState::DecisionPoint::handleInitialize() {
00382   EUROPA::TokenId cur = m_target->current(), active;
00383   EUROPA::eint date = m_target->now();
00384   
00385 
00386   m_choices.reset();
00387   m_choices.set(CREATE_DEFAULT);
00388   
00389 
00390   if( cur.isId() ) {
00391     active = cur->getActiveToken();
00392     if( active.isNoId() )
00393       active = cur;
00394     EUROPA::IntervalIntDomain const &end = active->end()->lastDomain();
00395     
00396     if( end.getLowerBound() <= date ) {
00397       m_choices.set(EXTEND_CURRENT, end.getUpperBound() > date);
00398     } else {
00399       // Already decided : extend_current
00400       m_choices.reset();
00401       m_idx = m_choices.size();
00402       return;
00403     }
00404   }
00405   // try to find the potential candidates
00406   std::list<EUROPA::TokenId> const &sequence = m_target->timeline()->getTokenSequence();
00407   std::list<EUROPA::TokenId>::const_iterator i=sequence.begin();
00408   if( active.isId() ) {
00409     // Last observation is active => find it 
00410     for(; sequence.end()!=i && active!=*i; ++i);
00411     // Then make the next token in the sequence the candidate
00412     if( sequence.end()!=i )
00413       ++i;
00414     m_cand_from = i;
00415     if( sequence.end()!=i )
00416       ++i;
00417     m_cand_to = i;
00418   } else {
00419     // No last observation active => search for tokens that can start now
00420     for(; sequence.end()!=i && (*i)->end()->lastDomain().getUpperBound()<=date; ++i);
00421     m_cand_from = i;
00422     for(; sequence.end()!=i && (*i)->start()->lastDomain().getUpperBound()<=date; ++i);
00423     m_cand_to = i;
00424   }
00425   // <m_cand_from, m_cand_to> reflects all the tokens in the sequence that can potentially overlap now
00426   m_choices.set(START_NEXT, m_cand_from!=m_cand_to);
00427   m_tok = m_cand_from;
00428   
00429   m_choices.set(CREATE_OTHER, !m_target->m_pred_names.empty());
00430   m_next_pred = m_target->m_pred_names.begin();
00431     
00432   // Initialize the decision index
00433   m_idx = 0;
00434   if( !m_choices[m_idx] ) 
00435     advance();
00436   m_prev_idx = m_idx;
00437 }

00441 
00442   m_prev_idx = m_idx;
00443   switch(m_idx) {

00446     break;
00447   case START_NEXT:

00460   default:
