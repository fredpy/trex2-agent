00128 void Goal::restrictTime(IntegerDomain const &s,
00129                      IntegerDomain const &d,
00130                      IntegerDomain const &e) {
00131   IntegerDomain::bound sLo, sHi, dLo, dHi, eLo, eHi;
00132   IntegerDomain::bound mLo, mHi, aLo, aHi;
00133 
00134   if( !( m_start.domain().intersect(s) && 
00135         m_duration.domain().intersect(d) &&
00136         m_end.domain().intersect(e) ) )
00137     throw PredicateException("Invalid time constraint on Goal");
00138   // Compute the new domains
00139   m_start.typedDomain<IntegerDomain>().getBounds(sLo, sHi);
00140   
00141   s.getBounds(aLo, aHi);
00142   // s = s inter start
00143   sLo = sLo.max(aLo);
00144   sHi = sHi.min(aHi);
00145 
00146   m_duration.typedDomain<IntegerDomain>().getBounds(dLo, dHi);
00147   d.getBounds(aLo, aHi);
00148   // d = d inter duration 
00149   dLo = dLo.max(aLo);
00150   dHi = dHi.min(aHi);
00151 
00152   m_end.typedDomain<IntegerDomain>().getBounds(eLo, eHi);
00153   e.getBounds(aLo, aHi);
00154   // e = e inter end
00155   eLo = eLo.max(aLo);
00156   eHi = eHi.min(aHi);
00157  
00158   // propagate end time 
00159   // e = e inter s+d
00160   eHi = eHi.min(sHi.plus(dHi, eHi));
00161   eLo = eLo.max(sLo.plus(dLo, eLo));
00162 
00163   if( eLo>eHi )
00164     throw PredicateException("end time is empty after propagation");
00165 
00166   // propagate start time 
00167   // s = s inter e-d
00168   sHi = sHi.min(eHi.minus(dLo, sHi));
00169   sLo = sLo.max(eLo.minus(dHi, sLo));
00170 
00171   if( sLo>sHi )
00172     throw PredicateException("start time is empty after propagation");
00173 
00174   // propagate duration 
00175   // d = d inter e-s
00176   dHi = dHi.min(eHi.minus(sLo, dHi));
00177   dLo = dLo.max(eLo.minus(sHi, dLo));
00178 
00179   if( dLo>dHi )
00180     throw PredicateException("duration is empty after propagation");
00181   try {
00182     // start = start inter s
00183     m_start.restrict(IntegerDomain(sLo, sHi));
00184   } catch( EmptyDomain const &e) {
00185     // this should not happen
00186     throw PredicateException(std::string("Goal: start progation: ")+e.what());
00187   }
00188   try {
00189     // duration = duration inter d
00190     m_duration.restrict(IntegerDomain(dLo, dHi));
00191   } catch( EmptyDomain const &e) {
00192     // this should not happen
00193     throw PredicateException(std::string("Goal: duration progation: ")+e.what());
00194   }
00195   try {
00196     // end = end inter e
00197     m_end.restrict(IntegerDomain(eLo, eHi));
00198   } catch( EmptyDomain const &e) {
00199     // this should not happen
00200     throw PredicateException(std::string("Goal: end progation: ")+e.what());
00201   }
