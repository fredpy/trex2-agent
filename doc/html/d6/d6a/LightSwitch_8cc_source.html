00141 bool Light::synchronize() {
00142   bool need_post = m_verbose;
00143 
00144   if( m_firstTick ) {
00145     setValue(m_on);
00146     need_post = false;
00147     m_firstTick = false;
00148     m_nextSwitch = getCurrentTick()+1;
00149   } else {
00150     TICK cur = getCurrentTick();
00151 
00152     if( m_nextSwitch<=cur ) {
00153       while( !m_pending.empty() )
00154        if( m_pending.front()->startsAfter(cur) ) {
00155          // it can start after cur
00156          if( m_pending.front()->startsBefore(cur) ) {
00157            
00158            // it can also starts before cur => it can be set to cur 
00159            if( brokenPred!=m_pending.front()->predicate() ) {
00160              setValue(downPred==m_pending.front()->predicate());
00161              need_post = false;
00162            } else {
00163              Observation obs(*m_pending.front());
00164              postObservation(obs);
00165            }
00166            m_nextSwitch = cur+m_pending.front()->getDuration().lowerBound().value();
00167            m_pending.pop_front();
00168          }
00169          // if we reached this point that mesans that the goal
00170          // necessarily starts in the future (or has been
00171          // processed)
00172          break;
00173        } else {
00174          // too late to execute => remove it
00175          m_pending.pop_front();
00176        }
00177     }
00178   }
00179   // check if I still need to post the ligh state
00180   if( need_post ) 
00181     postObservation(*m_light_state);
00182   // always succeed
00183   return true;
