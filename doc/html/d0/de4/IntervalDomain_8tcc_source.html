00094 std::istream &IntervalDomain<Ty, Cmp>::bound::read_from(std::istream &in) {
00095   // Attempt first to get [+-]inf
00096   bool is_pos;
00097   std::string buff;
00098   char c = in.peek();
00099   
00100   // check for optional sign character
00101   is_pos = c!='-';
00102   if( !is_pos || '+'==c ) {
00103     in.get(c);
00104     buff.insert(0, 1, c);
00105   }
00106   // look for "inf" case-insensitive
00107   static char const infstr[] = "INF";
00108   char const *i;
00109   i = infstr;
00110   while( *i && in.get(c) ) {
00111     buff.insert(0, 1, c);
00112     if( *i!=std::toupper(c) ) 
00113       break;
00114     ++i;
00115   }
00116   if( '\0'==*i ) {
00117     // success : set the bound to the corresponding set
00118     m_pos = is_pos;
00119     m_inf = true;
00120   } else {
00121     // failure : rollback and try to read Ty
00122     Ty val;
00123     for(size_t j=0; j<buff.length(); ++j )
00124       in.putback(buff[j]);
00125     in>>val;
00126     if( !in.bad() ) {
00127       // Alright it was a Ty value
00128       m_inf = false;
00129       m_value = val;
00130     }
00131   }
00132   return in;
