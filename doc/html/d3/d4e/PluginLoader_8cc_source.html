00083 void PluginLoader::load(Symbol const &name) {
00084   handle_map::iterator i = m_loaded.find(name);
00085   if( m_loaded.end()==i ) {
00086     bool found;
00087     std::string fileName = m_log->locate("lib"+name.str()+p_dlext(), found).string();
00088     if( found ) {
00089       m_log->syslog("plugin")<<"Loading "<<fileName;
00090       void *handle = p_dlopen(fileName.c_str(), RTLD_NOW);
00091       if( NULL==handle )
00092         throw PluginError(name, "Failed to load \""+name.str()+"\"");
00093       plugin_fn f_init = (plugin_fn)p_dlsym(handle, "initPlugin");
00094       if( NULL==f_init ) {
00095         PluginError err(name, "missing initPlugin");
00096         p_dlclose(handle);
00097         throw err;
00098       }
00099       m_loaded[name] = std::make_pair(handle, 1);
00100       // call to f_init : may throw an exception but
00101       //                  it is too be avoided
00102       f_init();
00103     } else 
00104       throw Exception("Unable to locate plugin "+name.str());
00105   } else {
00106     i->second.second += 1;
00107   }
00108 }
