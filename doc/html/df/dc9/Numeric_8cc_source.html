00057 void AbsValConstraint::handleExecute() {
00058   EUROPA::edouble a_lb, a_ub, v_lb, v_ub;
00059   // restrict m_abs based on m_val
00060   m_val.getBounds(v_lb, v_ub);
00061   
00062   if( v_lb>=0.0 ) {
00063     a_lb = v_lb;
00064     a_ub = v_ub;
00065   } else {
00066     if( v_ub>=0.0 )
00067       a_lb = 0.0;
00068     else 
00069       a_lb = std::min(std::abs(v_lb), std::abs(v_ub));
00070     a_ub = std::max(std::abs(v_lb), std::abs(v_ub));
00071   }
00072   m_abs.intersect(a_lb, a_ub);
00073 
00074   // constrain m_val based on m_abs
00075   m_abs.getBounds(a_lb, a_ub);
00076 
00077   if( a_lb==0.0 && m_val.isMember(-a_ub) && m_val.isMember(a_ub) )
00078     m_val.intersect(-a_ub, a_ub);
00079   if( ( m_val.isMember(a_lb) || m_val.isMember(a_ub) ) &&
00080       ( m_val.isMember(-a_lb) || m_val.isMember(-a_ub) ) )
00081     return;
00082   if( m_val.isMember(a_lb) || m_val.isMember(a_ub) )
00083     m_val.intersect(a_lb, a_ub);
00084   if( m_val.isMember(-a_lb) || m_val.isMember(-a_ub) )
00085     m_val.intersect(-a_ub, -a_lb);     
00086 }

00104   static EUROPA::IntervalDomain const positive(0.0, std::numeric_limits<EUROPA::edouble>::infinity());
00105   static EUROPA::edouble const acceptable_epsilon(1e-4);
00106 
00107   if( m_sqrt.isOpen() && m_val.isOpen() ) {
00108     m_sqrt.intersect(positive);
00109     m_val.intersect(positive);
00110     return;

00112     EUROPA::edouble val = m_val.getSingletonValue(), root;
00113     
00114     if( val<0.0 ) {
00115       m_val.intersect(positive);
00116       return;
00117     }
00118     root = std::sqrt(val);
00119     if( val>=std::numeric_limits<EUROPA::edouble>::infinity() )
00120       m_sqrt.intersect(root, std::numeric_limits<EUROPA::edouble>::infinity());
00121     else 
00122       m_sqrt.intersect(root, root);
00123     return;
00124   } else if( m_sqrt.isSingleton() ) {
00125     EUROPA::edouble root = m_sqrt.getSingletonValue(), val;
00126     
00127     if( root<0.0 ) {
00128       m_sqrt.intersect(positive);
00129       return;
00130     }
00131     val = root*root;
00132     if( val>=std::numeric_limits<EUROPA::edouble>::infinity() )
00133       m_val.intersect(root, std::numeric_limits<EUROPA::edouble>::infinity());
00134     else 
00135       m_val.intersect(val, val);
00136     return;
00137   }
00138   EUROPA::edouble v_min, v_max, r_min, r_max;
00139   
00140   for(bool done=false; !done; ) {
00141     done = true;
00142     m_val.getBounds(v_min, v_max);
00143     m_sqrt.getBounds(r_min, r_max);
00144     
00145     if( v_min<=0.0 )
00146       v_min = 0.0;
00147     if( r_min<=0.0 )
00148       r_min = 0.0;
00149     
00150     // compute upper bound of sqrt
00151     EUROPA::edouble max_r, min_r;
00152     if( v_max>=std::numeric_limits<EUROPA::edouble>::infinity() )
00153       max_r = v_max;
00154     else 
00155       max_r = std::sqrt(v_max);
00156     if( r_max-acceptable_epsilon > max_r )
00157       r_max = max_r;
00158     
00159     // compute lower bound of sqrt
00160     min_r = std::sqrt(v_min);
00161     if( r_min+acceptable_epsilon < min_r )
00162       r_min = min_r;
00163     if( m_sqrt.intersect(r_min, r_max) && m_sqrt.isEmpty() )
00164       return;
00165     
00166     EUROPA::edouble max_v, min_v;
00167     
00168     // compute upper bound of the square
00169     max_v = r_max*r_max;
00170     if( v_max-acceptable_epsilon > max_v )
00171       v_max = max_v;
00172     // compute the lower bound of the square
00173     min_v = r_min*r_min;
00174     if( v_min+acceptable_epsilon < min_v )
00175       v_min = min_v;
00176     
00177     if( m_val.intersect(v_min, v_max) ) {
00178       if( m_val.isEmpty() )
00179         return;
00180       // need to propagate the new domain to m_sqrt
00181       done = false;
00182     }
00183   }
