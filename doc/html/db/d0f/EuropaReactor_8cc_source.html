00445 bool EuropaReactor::hasWork() {
00446   setStream();

00448     syslog("ERROR")<<"Plan database is inconsistent.";
00449     return false;
00450   }
00451   if( planner()->isExhausted() ) {
00452     syslog("WARN")<<"Deliberation solver is exhausted.";
00453     return false;
00454   }
00455   if( !m_completed_this_tick ) {
00456     if( planner()->noMoreFlaws() ) {
00457       size_t steps = planner()->getStepCount();
00458       m_completed_this_tick = true;
00459       planner()->clear();
00460       archive();
00461       if( steps>0 ) {
00462         syslog()<<"Deliberation completed in "<<steps<<" steps.";
00463         logPlan("plan");
00464         getFuturePlan();
00465       }
00466       Assembly::external_iterator from(begin(), end()), to(end(), end());
00467       for(; to!=from; ++from) {
00468         TeleoReactor::external_iterator
00469         j=find_external((*from)->timeline()->getName().c_str());
00470         EUROPA::eint e_lo, e_hi;
00471         if( j.valid() && j->accept_goals() ) {
00472           IntegerDomain window = j->dispatch_window(getCurrentTick());
00473           IntegerDomain::bound lo = window.lowerBound(), hi = window.upperBound();
00474           e_lo = lo.value();
00475           if( hi.isInfinity() )
00476             e_hi = final_tick();
00477           else
00478             e_hi = hi.value();
00479           (*from)->do_dispatch(e_lo, e_hi);
00480         }
00481       }
00482     }
00483   }
00484   return !m_completed_this_tick;
00485 }
