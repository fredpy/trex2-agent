00369 bool Assembly::do_synchronize() {
00370   // 1) apply external observations
00371   for(external_iterator i=begin_external(); end_external()!=i; ++i) {
00372     if( !(*i)->commit() ) {
00373       debugMsg("trex:synch", "Failed to integrate external state of "<<(*i)->timeline()->toString());

00377 
00378   // 2) execute synchronizer
00379   if( !synchronizer()->solve() ) {
00380     debugMsg("trex:synch", "Failed to resolve synchronization for tick "<<now());
00381     return false;
00382   }
00383 
00384   // 3) apply internal observations
00385   for(internal_iterator i=begin_internal(); end_internal()!=i; ++i)
00386     (*i)->commit(); // It was consistent from synchronizer => never fail
00387 
00388   return true;

00399   for(EUROPA::TokenSet::const_iterator t=m_roots.begin(); m_roots.end()!=t; ++t) {
00400     bool is_past = (*t)->end()->baseDomain().getUpperBound()<=now();

00411           if( !(*t)->isInactive() )

00416     } else {
00417       if( !( (*t)->isInactive() || (*t)->isFact() ) )
