00060 void TREX::europa::details::restrict_bases(EUROPA::TokenId const &tok) {
00061   if( tok->isMerged() ) {    
00062     EUROPA::TokenId active = tok->getActiveToken();
00063     std::vector<EUROPA::ConstrainedVariableId> const &tvar = tok->getVariables();
00064     std::vector<EUROPA::ConstrainedVariableId> const &avar = active->getVariables();
00065     for(size_t i=1; i<tvar.size(); ++i) {
00066       size_t count=0;
00067       for(; !tvar[i]->isActive(); ++count ) 
00068         tvar[i]->undoDeactivation();
00069       tvar[i]->restrictBaseDomain(avar[i]->lastDomain());
00070       for(size_t j=0; j<count; ++j)
00071         tvar[i]->deactivate();
00072       avar[i]->handleBase(tvar[i]->baseDomain());
00073     }
00074   } else 
00075     tok->restrictBaseDomains();

00091 EUROPA::TokenId TREX::europa::details::parent_token(EUROPA::ConstrainedVariableId const &var) {

00093   
00094   if( ent.isId() ) {

00099       EUROPA::RuleInstanceId rule(ent);
