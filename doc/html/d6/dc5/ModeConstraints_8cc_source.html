00067 void CheckInternal::handleExecute() {
00068   if( NULL!=m_obj ) {
00069     Assembly &a = assembly();
00070     EUROPA::DataTypeId type = m_obj->getDataType();
00071 
00072     if( a.is_agent_timeline(type) ) {
00073       EUROPA::ObjectDomain const &dom(*m_obj);
00074       std::list<EUROPA::ObjectId> internals, 
00075        non_internals, vals = dom.makeObjectList();
00076 
00077       // Separate internal objects from the others
00078       for(std::list<EUROPA::ObjectId>::const_iterator o=vals.begin();
00079          vals.end()!=o; ++o) {
00080        if( a.internal(*o) )
00081          internals.push_back(*o);
00082        else 
00083          non_internals.push_back(*o);
00084       }
00085 
00086       if( !internals.empty() ) {
00087        if( non_internals.empty() ) {
00088          // Only internals
00089          m_test->intersect(true, true);
00090          return;
00091        } else if( m_test->isSingleton() ) {
00092          // Build the new domain based on test value
00093          std::auto_ptr<EUROPA::ObjectDomain> tmp;
00094          if( m_test->isMember(true) )
00095            tmp.reset(new EUROPA::ObjectDomain(type, internals));
00096          else 
00097            tmp.reset(new EUROPA::ObjectDomain(type, non_internals));
00098          // Restrict the domain of m_obj
00099          m_obj->intersect(*tmp);
00100          return;
00101        }
00102       }
00103     }
00104   }
00105   // If I am here then I knwo that te object is not internal
00106   m_test->intersect(false, false);
00107 }
