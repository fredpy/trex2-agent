00054                       double &UTMNorthing, double &UTMEasting, char* UTMZone)

00056 //converts lat/long to UTM coords.  Equations from USGS Bulletin 1532 
00057 //East Longitudes are positive, West longitudes are negative. 
00058 //North latitudes are positive, South latitudes are negative
00059 //Lat and Long are in decimal degrees
00060        //Written by Chuck Gantz- chuck.gantz@globalstar.com

00062        double a = ellipsoid[ReferenceEllipsoid].EquatorialRadius;

00064        double k0 = 0.9996;
00065 
00066        double LongOrigin;
00067        double eccPrimeSquared;
00068        double N, T, C, A, M;
00069        
00070 //Make sure the longitude is between -180.00 .. 179.9
00071        double LongTemp = (Long+180)-int((Long+180)/360)*360-180; // -180.00 .. 179.9;
00072 
00073        double LatRad = Lat*deg2rad;
00074        double LongRad = LongTemp*deg2rad;
00075        double LongOriginRad;
00076        int    ZoneNumber;
00077 
00078        ZoneNumber = int((LongTemp + 180)/6) + 1;
00079   
00080        if( Lat >= 56.0 && Lat < 64.0 && LongTemp >= 3.0 && LongTemp < 12.0 )
00081               ZoneNumber = 32;
00082 
00083   // Special zones for Svalbard
00084        if( Lat >= 72.0 && Lat < 84.0 ) 
00085        {
00086          if(      LongTemp >= 0.0  && LongTemp <  9.0 ) ZoneNumber = 31;
00087          else if( LongTemp >= 9.0  && LongTemp < 21.0 ) ZoneNumber = 33;
00088          else if( LongTemp >= 21.0 && LongTemp < 33.0 ) ZoneNumber = 35;
00089          else if( LongTemp >= 33.0 && LongTemp < 42.0 ) ZoneNumber = 37;
00090         }
00091        LongOrigin = (ZoneNumber - 1)*6 - 180 + 3;  //+3 puts origin in middle of zone
00092        LongOriginRad = LongOrigin * deg2rad;
00093 
00094        //compute the UTM Zone from the latitude and longitude
00095        sprintf(UTMZone, "%d%c", ZoneNumber, UTMLetterDesignator(Lat));
00096 
00097        eccPrimeSquared = (eccSquared)/(1-eccSquared);
00098 
00099        N = a/sqrt(1-eccSquared*sin(LatRad)*sin(LatRad));
00100        T = tan(LatRad)*tan(LatRad);
00101        C = eccPrimeSquared*cos(LatRad)*cos(LatRad);
00102        A = cos(LatRad)*(LongRad-LongOriginRad);
00103 
00104        M = a*((1     - eccSquared/4              - 3*eccSquared*eccSquared/64       - 5*eccSquared*eccSquared*eccSquared/256)*LatRad 
00105                             - (3*eccSquared/8    + 3*eccSquared*eccSquared/32       + 45*eccSquared*eccSquared*eccSquared/1024)*sin(2*LatRad)
00106                                                                + (15*eccSquared*eccSquared/256 + 45*eccSquared*eccSquared*eccSquared/1024)*sin(4*LatRad) 
00107                                                                - (35*eccSquared*eccSquared*eccSquared/3072)*sin(6*LatRad));
00108        
00109        UTMEasting = (double)(k0*N*(A+(1-T+C)*A*A*A/6
00110                                    + (5-18*T+T*T+72*C-58*eccPrimeSquared)*A*A*A*A*A/120)
00111                                    + 500000.0);
00112 
00113        UTMNorthing = (double)(k0*(M+N*tan(LatRad)*(A*A/2+(5-T+9*C+4*C*C)*A*A*A*A/24
00114                              + (61-58*T+T*T+600*C-330*eccPrimeSquared)*A*A*A*A*A*A/720)));

00116               UTMNorthing += 10000000.0; //10000000 meter offset for southern hemisphere

00156 //East Longitudes are positive, West longitudes are negative. 
00157 //North latitudes are positive, South latitudes are negative
00158 //Lat and Long are in decimal degrees. 
00159        //Written by Chuck Gantz- chuck.gantz@globalstar.com

00161        double k0 = 0.9996;

00163        double eccSquared = ellipsoid[ReferenceEllipsoid].eccentricitySquared;
00164        double eccPrimeSquared;
00165        double e1 = (1-sqrt(1-eccSquared))/(1+sqrt(1-eccSquared));
00166        double N1, T1, C1, R1, D, M;
00167        double LongOrigin;
00168        double mu, phi1, phi1Rad;
00169        double x, y;
00170        int ZoneNumber;
00171        char* ZoneLetter;
00172        int NorthernHemisphere; //1 for northern hemispher, 0 for southern
00173 
00174        x = UTMEasting - 500000.0; //remove 500,000 meter offset for longitude
00175        y = UTMNorthing;
00176 
00177        ZoneNumber = strtoul(UTMZone, &ZoneLetter, 10);
00178        if((*ZoneLetter - 'N') >= 0)
00179               NorthernHemisphere = 1;//point is in northern hemisphere
00180        else
00181        {
00182               NorthernHemisphere = 0;//point is in southern hemisphere
00183               y -= 10000000.0;//remove 10,000,000 meter offset used for southern hemisphere
00184        }
00185 
00186        LongOrigin = (ZoneNumber - 1)*6 - 180 + 3;  //+3 puts origin in middle of zone
00187 
00188        eccPrimeSquared = (eccSquared)/(1-eccSquared);
00189 
00190        M = y / k0;
00191        mu = M/(a*(1-eccSquared/4-3*eccSquared*eccSquared/64-5*eccSquared*eccSquared*eccSquared/256));
00192 
00193        phi1Rad = mu  + (3*e1/2-27*e1*e1*e1/32)*sin(2*mu) 
00194                             + (21*e1*e1/16-55*e1*e1*e1*e1/32)*sin(4*mu)
00195                             +(151*e1*e1*e1/96)*sin(6*mu);
00196        phi1 = phi1Rad*rad2deg;
00197 
00198        N1 = a/sqrt(1-eccSquared*sin(phi1Rad)*sin(phi1Rad));
00199        T1 = tan(phi1Rad)*tan(phi1Rad);
00200        C1 = eccPrimeSquared*cos(phi1Rad)*cos(phi1Rad);
00201        R1 = a*(1-eccSquared)/pow(1-eccSquared*sin(phi1Rad)*sin(phi1Rad), 1.5);
00202        D = x/(N1*k0);
00203 
00204        Lat = phi1Rad - (N1*tan(phi1Rad)/R1)*(D*D/2-(5+3*T1+10*C1-4*C1*C1-9*eccPrimeSquared)*D*D*D*D/24
00205                                    +(61+90*T1+298*C1+45*T1*T1-252*eccPrimeSquared-3*C1*C1)*D*D*D*D*D*D/720);
00206        Lat = Lat * rad2deg;
00207 
00208        Long = (D-(1+2*T1+C1)*D*D*D/6+(5-2*C1+28*T1-3*C1*C1+8*eccPrimeSquared+24*T1*T1)
00209                                    *D*D*D*D*D/120)/cos(phi1Rad);
00210        Long = LongOrigin + Long * rad2deg;
00211 
00212 }
